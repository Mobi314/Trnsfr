Start of Previous Quarter = 
VAR SelectedDate = SELECTEDVALUE('Current Status'[Date], TODAY())
VAR FiscalYear = YEAR(SelectedDate) - IF(MONTH(SelectedDate) < 4, 1, 0)
VAR Quarter = 
    SWITCH(TRUE(),
        MONTH(SelectedDate) >= 4 && MONTH(SelectedDate) <= 6, "Q1",
        MONTH(SelectedDate) >= 7 && MONTH(SelectedDate) <= 9, "Q2",
        MONTH(SelectedDate) >= 10 && MONTH(SelectedDate) <= 12, "Q3",
        "Q4"
    )
RETURN
    SWITCH(Quarter,
        "Q1", DATE(FiscalYear - 1, 10, 1), // If current is Q1, previous is Q4 of the previous fiscal year
        "Q2", DATE(FiscalYear, 4, 1),     // If current is Q2, previous is Q1 of the same fiscal year
        "Q3", DATE(FiscalYear, 7, 1),     // If current is Q3, previous is Q2 of the same fiscal year
        DATE(FiscalYear, 10, 1)           // If current is Q4, previous is Q3 of the same fiscal year
    )

Start of Previous Quarter = 
VAR SelectedDate = SELECTEDVALUE('Current Status'[Date], TODAY())
VAR YearAdjustment = IF(MONTH(SelectedDate) < 4, -1, 0)
VAR CurrentFiscalYear = YEAR(SelectedDate) + YearAdjustment
VAR CurrentQuarter = 
    SWITCH(
        TRUE(),
        MONTH(SelectedDate) >= 4 AND MONTH(SelectedDate) <= 6, "Q1",
        MONTH(SelectedDate) >= 7 AND MONTH(SelectedDate) <= 9, "Q2",
        MONTH(SelectedDate) >= 10 AND MONTH(SelectedDate) <= 12, "Q3",
        "Q4"  // This accounts for January to March
    )
RETURN
    SWITCH(
        CurrentQuarter,
        "Q1", DATE(CurrentFiscalYear - 1, 10, 1),  // Start of Q4 in the previous fiscal year
        "Q2", DATE(CurrentFiscalYear, 4, 1),       // Start of Q1 in the current fiscal year
        "Q3", DATE(CurrentFiscalYear, 7, 1),       // Start of Q2 in the current fiscal year
        DATE(CurrentFiscalYear, 10, 1)             // Start of Q3 in the current fiscal year for "Q4"
    )

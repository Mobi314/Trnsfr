Start of Previous Quarter = 
VAR CurrentQ = [Current Quarter]
VAR FiscalYearStart = DATE(YEAR(TODAY()) - (MONTH(TODAY()) < 4), 4, 1)
VAR Q1Start = FiscalYearStart
VAR Q2Start = DATE(YEAR(FiscalYearStart), 7, 1)
VAR Q3Start = DATE(YEAR(FiscalYearStart), 10, 1)
VAR Q4Start = DATE(YEAR(FiscalYearStart) + 1, 1, 1)
RETURN
    SWITCH(CurrentQ,
        "Q1", DATE(YEAR(FiscalYearStart) - 1, 10, 1),  // If current quarter is Q1, previous is Q4 of last year
        "Q2", Q1Start,  // If current quarter is Q2, previous is Q1
        "Q3", Q2Start,  // If current quarter is Q3, previous is Q2
        "Q4", Q3Start   // If current quarter is Q4, previous is Q3
    )
